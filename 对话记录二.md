### 问题 1：Windows 终端错误（严重程度：中）

**问题分析：**
- **根本原因**：node-pty 在 Windows 上的行为与 Linux/macOS 不同。当 shell 进程退出后，pty 实例可能未及时清理，导致后续 resize 操作失败
- **影响范围**：仅影响 Windows 服务端，Linux/macOS 服务端正常
- **实际影响**：用户如果尝试调整已关闭终端的窗口大小会看到错误提示，但终端已关闭不影响其他功能

**解决方案建议：**
```typescript
// server/src/services/terminalService.ts 或类似文件
class TerminalSession {
  private pty: IPty | null = null;
  private isExited: boolean = false;

  public resize(cols: number, rows: number): void {
    // 添加状态检查
    if (this.isExited || !this.pty) {
      console.warn(`Attempted to resize exited terminal session ${this.id}`);
      return; // 静默失败，不抛出错误
    }
    
    try {
      this.pty.resize(cols, rows);
    } catch (error) {
      // 捕获错误，如果 pty 已退出则标记状态
      if (error.message?.includes('already exited')) {
        this.isExited = true;
        this.pty = null;
      } else {
        throw error;
      }
    }
  }
  
  public onExit(): void {
    this.isExited = true;
    this.pty = null;
    // 其他清理逻辑...
  }
}
```

```typescript
function smartTruncate(text: string, maxLength: number): string {
  if (text.length <= maxLength) return text;
  
  // 找到最后一个单词边界
  const truncated = text.substring(0, maxLength);
  const lastSpaceIndex = truncated.lastIndexOf(' ');
  
  if (lastSpaceIndex > maxLength * 0.5) {
    return truncated.substring(0, lastSpaceIndex) + '...';
  }
  
  return truncated + '...';
}
```
